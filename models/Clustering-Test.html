  • No data leakage

    • MINKGEM vs MINK4575 (overall vs bracket) keep one

Variable choice

  • Keep the signal, drop redundancies:

    • Products (household): APERSAUT, ABRAND, AWAPART

    • Local context: PPERSAUT, PBRAND, PWAPART

    • Affluence: MINKGEM (or keep MINK4575, but not both), MKOOPKLA

    • Housing: MHKOOP (drop MHHUUR)

    • Education/status: MOPLHOOG (drop MOPLLAAG), MBERHOOG (or MSKA, pick one)

    • Segmentation: MOSTYPE (drop MOSHOOFD)

Clustering with Gower + PAM

# --- 1) Choose a compact, interpretable variable set (from your IG ranking) ---
shortlist <- c(
  "APERSAUT","ABRAND","AWAPART",     # household product holdings
  "PPERSAUT","PBRAND","PWAPART",     # postcode prevalence of those products
  "MINKGEM","MKOOPKLA",              # affluence / purchase power
  "MHKOOP","MOPLHOOG","MBERHOOG",    # housing, education, status
  "MOSTYPE"                          # customer main type (nominal)
)
shortlist <- intersect(shortlist, names(training_set))  # safety

clust_df <- training_set[, c(shortlist, "CARAVAN")]

# Set obvious nominals to factor for Gower distance
if (!is.factor(clust_df$MOSTYPE)) clust_df$MOSTYPE <- factor(clust_df$MOSTYPE)
if (!is.factor(clust_df$MHKOOP))  clust_df$MHKOOP  <- factor(clust_df$MHKOOP)
# --- 2) Quick redundancy check (Spearman on numeric-only) ---------------------
num_cols <- names(clust_df)[sapply(clust_df, is.numeric)]
if (length(num_cols) > 1) {
  corr_mat <- suppressWarnings(cor(clust_df[, num_cols], method = "spearman", use = "pairwise.complete.obs"))
  # Find highly correlated pairs (|r| > .90)
  hi_pairs <- which(abs(corr_mat) > 0.90 & upper.tri(corr_mat), arr.ind = TRUE)
  to_drop  <- character(0)
  if (nrow(hi_pairs) > 0) {
    # Drop the second variable of each pair
    to_drop <- unique(colnames(corr_mat)[hi_pairs[,2]])
  }
  if (length(to_drop)) {
    message("Dropping highly redundant numeric vars (>0.90): ", paste(to_drop, collapse = ", "))
    shortlist <- setdiff(shortlist, to_drop)
    clust_df  <- clust_df[, c(shortlist, "CARAVAN")]
  }
}
## Dropping highly redundant numeric vars (>0.90): PPERSAUT, PBRAND, PWAPART
# --- 3) Gower distance (handles mixed types) ----------------------------------
gower_dist <- daisy(clust_df[, shortlist], metric = "gower")
# --- 4) Choose k by average silhouette ----------------------------------------
k_grid <- 3:6
sil_tbl <- map_df(k_grid, function(k){
  fit <- pam(gower_dist, k = k, diss = TRUE)
  tibble(k = k, avg_sil = fit$silinfo$avg.width)
})

# Helper: find the block of lines for a given VAR and parse "code -> text"
parse_value_labels <- function(var){
  # find header line (e.g., " 12 MOSTYPE ...")
  hdr <- which(str_detect(lines, paste0("^\\s*\\d+\\s+", var, "\\b")))
  if (length(hdr) == 0) return(tibble(code = integer(), label = character()))
  start <- hdr[1] + 1
  # next var header (or end of file)
  next_hdr <- which(str_detect(lines[(start+0):length(lines)], "^\\s*\\d+\\s+[A-Z][A-Z0-9]+\\b"))
  end <- if (length(next_hdr)) start + next_hdr[1] - 2 else length(lines)

  block <- lines[start:end]

  # Accept common formats like "1: text", "1 = text", "1  text"
  m <- str_match(block, "^\\s*(\\d{1,3})\\s*[:=\\.-]?\\s*(.+\\S)\\s*$")
  out <- tibble(code = as.integer(m[,2]), label = m[,3]) %>%
    filter(!is.na(code), !is.na(label)) %>%
    distinct(code, .keep_all = TRUE) %>%
    arrange(code)
  out
}

  p_lolli <- ggplot(top5, aes(x = reorder(feature_lbl, abs(z)), y = z)) +
    geom_segment(aes(xend = reorder(feature_lbl, abs(z)), y = 0, yend = z), linewidth = 0.6) +
    geom_point(size = 2) +
    coord_flip() +
    facet_wrap(~ cluster_name, ncol = 2, scales = "free_y") +
    labs(title = "Top 5 differentiators per cluster (z vs overall mean)",
         x = "", y = "z-score") +
    theme_minimal(base_size = 12)
  print(p_lolli)
}
